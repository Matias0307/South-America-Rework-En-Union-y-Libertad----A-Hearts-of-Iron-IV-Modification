# Fascist Event 6 | Hitler arrives in Argentina
country_event = {
	id = argentina.fascist.6
	title = argentina.fascist.6.t 
	desc = argentina.fascist.6.d 
	picture = GFX_ARG_hitler_arrives

	fire_only_once = yes 

	trigger = {
		original_tag = ARG 
		has_completed_focus = ARG_create_the_CNU
		GER = { 
			OR = {
				has_civil_war = yes 
				NOT = { has_government = fascism }
				exists = no 
			}
		}
		has_country_leader = {
			character = ARG_walter_darre
			ruling_only = yes 
		}
		NOT = { has_global_flag = ARG_hitler_in_argentina_flag }
	}

	immediate = {
		hidden_effect = {
			random_country_with_original_tag = {
				original_tag_to_check = GER
				if = {
					limit = {
						has_country_leader = {
							character = GER_adolf_hitler
							ruling_only = yes 
						}
					}
					retire_character = GER_adolf_hitler
				}
			}
			set_global_flag = ARG_hitler_in_argentina_flag
		}
	}

	option = { # let him in 
		name = argentina.fascist.6.a 
		add_war_support = 0.01
	}

	option = { # give him role in government
		name = argentina.fascist.6.b 
		ai_chance = {
			base = 40
		}
		add_political_power = 10
		custom_effect_tooltip = available_political_advisor
		show_ideas_tooltip = ARG_adolf_hitler_idea
		set_country_flag = ARG_allowed_hitler_government_role_flag
	}

	option = { # he's not stepping foot in this country
		name = argentina.fascist.6.c 
		add_stability = 0.01
	}
}

## Posadist Ambition Triggerer - Either Rebuild the Economy, deal with the Oligarchs, or improve national infrastructure
country_event = {
	id = argentina.posadistambitions.102 
	
	hidden = yes 

	fire_only_once = yes 

	trigger = {
		original_tag = ARG 
		has_global_flag = ARG_civil_war_concluded
		has_country_leader = {
			character = ARG_juan_posadas
			ruling_only = yes 
		}
		has_civil_war = no 
		all_owned_state = { ## Makes sure the player has not lost any state 
			is_fully_controlled_by = ROOT
		}
		NOT = { has_country_flag = ARG_intermediate_ambition_triggered }
	}

	immediate = {
		random_list = {
			33 = {
				country_event = {
					id = argentina.posadistambitions.2 ## Rebuild the Economy
					days = 14
					random_days = 7
				}
			}
			33 = {
				modifier = {
					factor = 0
					NOT = {
						has_idea = ARG_agitated_oligarchs_idea
					}
				}
				country_event = {
					id = argentina.posadistambitions.3 ## Deal with the Oligarchs
					days = 30
					random_days = 14
				}
			}
			34 = {
				country_event = {
					id = argentina.posadistambitions.4 ## Improve National Infrastructure
					days = 14
					random_days = 7
				}
			}
		}
		set_country_flag = ARG_intermediate_ambition_triggered
		country_event = {
			id = argentina.posadistambitions.5
			days = 215
			random_days = 110
		}
		country_event = {
			id = argentina.posadistambitions.9
			days = 400
			random_days = 150
		}
	}
}
