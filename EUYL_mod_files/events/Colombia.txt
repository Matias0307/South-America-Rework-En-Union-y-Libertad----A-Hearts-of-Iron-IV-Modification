#Guerra Civil Conservadora

country_event = {
	id = colombia.25
	title = colombia.25.title
	desc = colombia.25.desc
	
	picture = GFX_report_event_yugoslavia_partisans

	is_triggered_only = yes

	#civil war starts
	option = {
		name = colombia.25.a
		custom_effect_tooltip = colombia_25_tt
		hidden_effect = {
			set_capital = { state = 306 }
			if = {
				limit = {
					NOT = { has_template = "Milicias" }
				}
				division_template = { 
					name = "Milicias" 
					division_names_group = SPAN_INF_02 
					is_locked = yes 
					regiments = { 
						infantry = { x = 0 y = 0 }  
						infantry = { x = 0 y = 1 } 
						infantry = { x = 0 y = 2 } 
					}
				}
			}
			start_civil_war = {
				ideology = fascism
				size = 0.2
				capital = 1014
				states = { 1014 1018 1019 }
			}
			remove_ideas = COL_oposicion
			every_controlled_state = {
				create_unit = {
					division = "name = \"Milicias\" division_template = \"Milicias\" start_experience_factor = 0.02 start_equipment_factor = 0.9" 
					owner = ROOT
					count = 2
				}
			}
		}
		hidden_effect = {
			random_other_country = {
				limit = {
					original_tag = COL
					has_war_with = ROOT
				}
				every_controlled_state = {
					create_unit = {
						division = "name = \"Milicias\" division_template = \"Milicias\" start_experience_factor = 0.02 start_equipment_factor = 0.9" 
						owner = ROOT
						count = 2
					}
				}

				remove_ideas = COL_oposicion

				set_party_name = {
					ideology = fascism
					name = "Partido Conservador"
					long_name = "Partido Conservador"
				}

				if = {
					limit = {
						has_government = fascism
					}

					set_cosmetic_tag = COL_azul_war

					create_country_leader = {
						name = "Laureano Gómez"
						desc = "POLITICS_LAUREANO_GOMEZ_DESC"
						picture = "gfx/leaders/COL/Portrait_Laureano_Gomez.dds"
						expire = "1965.1.1"
						ideology = rexism
						traits = {
							conservative_grandee
						}
					}
				}
			}
			random_other_country = {
				limit = {
					original_tag = COL
					has_war_with = ROOT
					has_government = fascism
				}
				every_controlled_state = {
					create_unit = {
						division = "name = \"Milicias\" division_template = \"Milicias\" start_experience_factor = 0.02 start_equipment_factor = 0.9" 
						owner = PREV
						count = 2
					}
				}
				save_event_target_as = conservador_civil_war
				every_unit_leader = {
					set_nationality = COL
				}
			}
			COL = {
				every_unit_leader = {
					limit = {
						OR = {
							has_id = 10001
						}
					}
					set_nationality = event_target:conservador_civil_war
				}
			}
		}
	}
}
